  ```c++
  uint64 insert_zero_bit(uint64 value, int pos) {
      uint64 bottom_mask = (1u64 << pos) - 1;
      uint64 top_mask = ~bottom_mask;
  
      uint64 bottom_bits = value & bottom_mask;
      uint64 top_bits = value & top_mask;
      return bottom_bits | (top_bits << 1);
  }
